## 2. HTTP Server: `src/router/server.py`

Minimal threaded HTTP server exposing the router's functionality over HTTP.

```python
"""Mesh router HTTP server.

Exposes router endpoints over HTTP for worker communication.
Uses stdlib ThreadingHTTPServer — zero external dependencies for serving.
Integrates with systemd via sd_notify for Type=notify watchdog support.
"""
import json
import logging
import os
import signal
import sys
from http.server import BaseHTTPRequestHandler, ThreadingHTTPServer
from datetime import datetime, timezone

from src.router.db import RouterDB
from src.router.heartbeat import HeartbeatTracker
from src.router.scheduler import Scheduler
from src.router.recovery import RecoveryManager
from src.router.bridge.transport import InProcessTransport

logger = logging.getLogger("mesh.server")


class MeshRouterHandler(BaseHTTPRequestHandler):
    """HTTP request handler for mesh router endpoints."""

    # Reference to shared server state (set by MeshRouterServer)
    router_state = None

    def do_GET(self):
        if self.path == "/health":
            self._handle_health()
        elif self.path.startswith("/tasks/next"):
            self._handle_task_poll()
        else:
            self._send_json(404, {"error": "not_found"})

    def do_POST(self):
        if self.path == "/events":
            self._handle_events()
        elif self.path == "/heartbeat":
            self._handle_heartbeat()
        elif self.path == "/register":
            self._handle_register()
        elif self.path == "/tasks/complete":
            self._handle_task_complete()
        elif self.path == "/tasks/fail":
            self._handle_task_fail()
        else:
            self._send_json(404, {"error": "not_found"})

    # --- Endpoint implementations ---

    def _handle_health(self):
        """GET /health — liveness check. No auth required (internal wg0 network only)."""
        state = self.server.router_state
        worker_count = len(state["db"].get_workers())
        queue_depth = len(state["db"].get_tasks_by_status("queued"))
        self._send_json(200, {
            "status": "healthy",
            "workers": worker_count,
            "queue_depth": queue_depth,
            "uptime_s": (datetime.now(timezone.utc) - state["start_time"]).total_seconds(),
        })

    def _handle_task_poll(self):
        """GET /tasks/next?worker_id=X — short-poll for next assigned task.

        Returns 200 + task JSON if a task is assigned, or 204 if none.
        MVP uses short-polling (worker polls every 2s). Long-polling deferred
        to v2 since stdlib HTTP server doesn't support async hold-open cleanly.
        """
        from urllib.parse import urlparse, parse_qs
        query = parse_qs(urlparse(self.path).query)
        worker_id = query.get("worker_id", [None])[0]
        if not worker_id:
            self._send_json(400, {"error": "missing worker_id"})
            return

        # Check auth token
        if not self._check_auth():
            return

        state = self.server.router_state
        db = state["db"]

        # Find tasks assigned to this worker that are in "assigned" status
        tasks = db.get_tasks_by_worker(worker_id, status="assigned")
        if tasks:
            task = tasks[0]  # Oldest assigned task
            self._send_json(200, task.model_dump(mode="json"))
        else:
            self._send_json(204, None)

    def _handle_events(self):
        """POST /events — receive CloudEvent JSON from bridge.

        Uses InProcessTransport to parse CloudEvent and write TaskEvent to DB.
        InProcessTransport is the correct abstraction here: it handles CloudEvent
        deserialization, idempotency key checking, and DB insertion — the same
        logic needed whether events arrive in-process or over HTTP.
        """
        if not self._check_auth():
            return
        body = self._read_body()
        if body is None:
            return
        try:
            json.loads(body)  # Validate JSON first
            transport = self.server.router_state["transport"]
            success = transport.send(body)
            if success:
                self._send_json(202, {"status": "accepted"})
            else:
                self._send_json(409, {"status": "duplicate_or_error"})
        except json.JSONDecodeError:
            self._send_json(400, {"error": "invalid_json"})

    def _handle_heartbeat(self):
        """POST /heartbeat — receive worker heartbeat."""
        if not self._check_auth():
            return
        body = self._read_body()
        if body is None:
            return
        try:
            data = json.loads(body)
            worker_id = data.get("worker_id")
            if not worker_id:
                self._send_json(400, {"error": "missing worker_id"})
                return
            tracker = self.server.router_state["heartbeat"]
            tracker.record_heartbeat(worker_id)
            self._send_json(200, {"status": "ok"})
        except json.JSONDecodeError:
            self._send_json(400, {"error": "invalid_json"})

    def _handle_register(self):
        """POST /register — worker registration."""
        if not self._check_auth():
            return
        body = self._read_body()
        if body is None:
            return
        try:
            data = json.loads(body)
            db = self.server.router_state["db"]
            from src.router.models import Worker
            worker = Worker(**data)
            db.upsert_worker(worker)
            self._send_json(201, {"status": "registered", "worker_id": worker.worker_id})
        except json.JSONDecodeError:
            self._send_json(400, {"error": "invalid_json"})
        except (ValueError, TypeError, KeyError) as e:
            self._send_json(400, {"error": f"invalid_worker_data: {type(e).__name__}"})

    def _handle_task_complete(self):
        """POST /tasks/complete — worker reports task completion."""
        if not self._check_auth():
            return
        body = self._read_body()
        if body is None:
            return
        try:
            data = json.loads(body)
            scheduler = self.server.router_state["scheduler"]
            scheduler.report_completion(
                task_id=data["task_id"],
                worker_id=data["worker_id"],
                result=data.get("result"),
            )
            self._send_json(200, {"status": "completed"})
        except json.JSONDecodeError:
            self._send_json(400, {"error": "invalid_json"})
        except KeyError as e:
            self._send_json(400, {"error": f"missing_field: {e}"})
        except (ValueError, RuntimeError) as e:
            self._send_json(409, {"error": f"transition_error: {type(e).__name__}"})

    def _handle_task_fail(self):
        """POST /tasks/fail — worker reports task failure."""
        if not self._check_auth():
            return
        body = self._read_body()
        if body is None:
            return
        try:
            data = json.loads(body)
            scheduler = self.server.router_state["scheduler"]
            scheduler.report_failure(
                task_id=data["task_id"],
                worker_id=data["worker_id"],
                error=data.get("error", "unknown"),
            )
            self._send_json(200, {"status": "failed_recorded"})
        except json.JSONDecodeError:
            self._send_json(400, {"error": "invalid_json"})
        except KeyError as e:
            self._send_json(400, {"error": f"missing_field: {e}"})
        except (ValueError, RuntimeError) as e:
            self._send_json(409, {"error": f"transition_error: {type(e).__name__}"})

    # --- Helpers ---

    def _check_auth(self) -> bool:
        """Validate bearer token if configured."""
        expected = self.server.router_state.get("auth_token")
        if not expected:
            return True  # No auth configured
        auth_header = self.headers.get("Authorization", "")
        if auth_header == f"Bearer {expected}":
            return True
        self._send_json(401, {"error": "unauthorized"})
        return False

    def _read_body(self) -> str | None:
        """Read request body. Returns None on error (already sent response)."""
        content_length = int(self.headers.get("Content-Length", 0))
        if content_length == 0:
            self._send_json(400, {"error": "empty_body"})
            return None
        if content_length > 1_000_000:  # 1MB max
            self._send_json(413, {"error": "body_too_large"})
            return None
        return self.rfile.read(content_length).decode("utf-8")

    def _send_json(self, status: int, data: dict | None):
        """Send JSON response."""
        self.send_response(status)
        if data is not None:
            body = json.dumps(data).encode("utf-8")
            self.send_header("Content-Type", "application/json")
            self.send_header("Content-Length", str(len(body)))
            self.end_headers()
            self.wfile.write(body)
        else:
            self.send_header("Content-Length", "0")
            self.end_headers()

    def log_message(self, format, *args):
        """Route access logs to logger instead of stderr."""
        logger.info(format, *args)


def run_server(
    host: str = "0.0.0.0",
    port: int = 8780,
    db_path: str = "/var/lib/mesh-router/router.db",
    auth_token: str | None = None,
):
    """Start the mesh router HTTP server.

    Initializes DB, recovery, scheduler, and heartbeat tracker,
    then serves HTTP requests. Sends sd_notify for systemd integration.
    """
    # Initialize components
    db = RouterDB(db_path)
    recovery = RecoveryManager(db)
    recovery.run_recovery()  # Requeue expired leases on startup

    heartbeat = HeartbeatTracker(db)
    scheduler = Scheduler(db, heartbeat_tracker=heartbeat)
    transport = InProcessTransport(db)
    start_time = datetime.now(timezone.utc)

    server = ThreadingHTTPServer((host, port), MeshRouterHandler)
    server.router_state = {
        "db": db,
        "heartbeat": heartbeat,
        "scheduler": scheduler,
        "transport": transport,
        "auth_token": auth_token,
        "start_time": start_time,
    }

    # Graceful shutdown on SIGTERM
    def handle_shutdown(signum, frame):
        logger.info("Shutting down mesh router...")
        server.shutdown()
    signal.signal(signal.SIGTERM, handle_shutdown)
    signal.signal(signal.SIGINT, handle_shutdown)

    # Notify systemd we're ready + start watchdog thread
    try:
        import sdnotify
        n = sdnotify.SystemdNotifier()
        n.notify("READY=1")
        logger.info("sd_notify: READY=1")

        # Periodic watchdog notification (WatchdogSec=30 in unit file)
        import threading
        def watchdog_loop():
            while True:
                n.notify("WATCHDOG=1")
                time.sleep(10)  # Notify every 10s (WatchdogSec=30)
        wd_thread = threading.Thread(target=watchdog_loop, daemon=True)
        wd_thread.start()
    except ImportError:
        logger.debug("sdnotify not available, skipping sd_notify")

    logger.info("Mesh router listening on %s:%d", host, port)
    server.serve_forever()


if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(name)s %(levelname)s %(message)s")
    port = int(os.environ.get("MESH_ROUTER_PORT", "8780"))
    db_path = os.environ.get("MESH_DB_PATH", "/var/lib/mesh-router/router.db")
    auth_token = os.environ.get("MESH_AUTH_TOKEN")
    run_server(port=port, db_path=db_path, auth_token=auth_token)
```

### Key design choices:
- **stdlib only** for HTTP serving (no Flask, no uvicorn)
- **ThreadingHTTPServer** for concurrent request handling
- **sd_notify** for systemd Type=notify integration
- **Bearer token auth** on all mutating endpoints
- **1MB max body** as safety limit
- **Graceful shutdown** via SIGTERM handler

## 3. Worker Client: `src/router/worker_client.py`

Worker-side entry point that registers, heartbeats, and polls for tasks.

```python
"""Mesh worker client.

Entry point for worker processes. Handles:
- Registration with the router
- Periodic heartbeat (5s interval)
- Task polling (GET /tasks/next)
- Task execution delegation (invoke CLI)
- Result reporting (complete/fail)
"""
import json
import logging
import os
import signal
import threading
import time
from dataclasses import dataclass

import requests

logger = logging.getLogger("mesh.worker")


@dataclass
class WorkerConfig:
    worker_id: str
    router_url: str  # e.g. "http://10.x.x.1:8780"
    cli_type: str  # "claude" | "codex" | "gemini"
    account_profile: str  # "work" | "clientA" | ...
    auth_token: str | None = None
    heartbeat_interval: float = 5.0
    poll_interval: float = 2.0
    capabilities: list[str] | None = None


class MeshWorker:
    """Worker client that communicates with the mesh router."""

    def __init__(self, config: WorkerConfig):
        self.config = config
        self._running = False
        self._heartbeat_thread: threading.Thread | None = None
        self._session = requests.Session()
        if config.auth_token:
            self._session.headers["Authorization"] = f"Bearer {config.auth_token}"
        self._session.headers["Content-Type"] = "application/json"

    def start(self):
        """Register with router, start heartbeat, begin polling."""
        self._running = True
        self._register()
        self._start_heartbeat()
        self._poll_loop()

    def stop(self):
        """Graceful shutdown."""
        self._running = False
        if self._heartbeat_thread:
            self._heartbeat_thread.join(timeout=10)

    def _register(self):
        """POST /register with worker metadata."""
        url = f"{self.config.router_url}/register"
        payload = {
            "worker_id": self.config.worker_id,
            "machine": os.environ.get("HOSTNAME", "unknown"),
            "cli_type": self.config.cli_type,
            "account_profile": self.config.account_profile,
            "capabilities": self.config.capabilities or ["code", "tests", "refactor"],
            "status": "idle",
            "concurrency": 1,
        }
        try:
            resp = self._session.post(url, json=payload, timeout=5)
            resp.raise_for_status()
            logger.info("Registered as %s", self.config.worker_id)
        except requests.RequestException as e:
            logger.error("Registration failed: %s", e)
            raise

    def _start_heartbeat(self):
        """Start background heartbeat thread."""
        def heartbeat_loop():
            url = f"{self.config.router_url}/heartbeat"
            while self._running:
                try:
                    self._session.post(
                        url,
                        json={"worker_id": self.config.worker_id},
                        timeout=3,
                    )
                except requests.RequestException as e:
                    logger.warning("Heartbeat failed: %s", e)
                time.sleep(self.config.heartbeat_interval)

        self._heartbeat_thread = threading.Thread(target=heartbeat_loop, daemon=True)
        self._heartbeat_thread.start()

    def _poll_loop(self):
        """Main loop: poll for tasks, execute, report."""
        url = f"{self.config.router_url}/tasks/next?worker_id={self.config.worker_id}"
        while self._running:
            try:
                resp = self._session.get(url, timeout=10)
                if resp.status_code == 200:
                    task = resp.json()
                    self._execute_task(task)
                elif resp.status_code == 204:
                    pass  # No tasks available
                else:
                    logger.warning("Poll returned %d", resp.status_code)
            except requests.RequestException as e:
                logger.warning("Poll failed: %s", e)
            time.sleep(self.config.poll_interval)

    def _execute_task(self, task: dict):
        """Execute a task via CLI invocation.

        This is the integration point with CCS / CLI tools.
        For v1, logs task receipt and reports success.
        Full CLI integration to be wired in production.
        """
        task_id = task["task_id"]
        logger.info("Executing task %s: %s", task_id, task.get("title", "untitled"))

        try:
            # TODO: Wire CLI invocation (CCS profile + command dispatch)
            # For now: placeholder that logs and reports success
            result = {"output": f"Task {task_id} executed by {self.config.worker_id}"}

            # Report completion
            self._session.post(
                f"{self.config.router_url}/tasks/complete",
                json={"task_id": task_id, "worker_id": self.config.worker_id, "result": result},
                timeout=5,
            )
            logger.info("Task %s completed", task_id)
        except Exception as e:
            logger.error("Task %s failed: %s", task_id, e)
            try:
                self._session.post(
                    f"{self.config.router_url}/tasks/fail",
                    json={"task_id": task_id, "worker_id": self.config.worker_id, "error": str(e)},
                    timeout=5,
                )
            except requests.RequestException:
                logger.error("Failed to report failure for task %s", task_id)


def run_worker():
    """Entry point for mesh-worker service."""
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(name)s %(levelname)s %(message)s")

    config = WorkerConfig(
        worker_id=os.environ.get("MESH_WORKER_ID", "ws-unknown-01"),
        router_url=os.environ.get("MESH_ROUTER_URL", "http://localhost:8780"),
        cli_type=os.environ.get("MESH_CLI_TYPE", "claude"),
        account_profile=os.environ.get("MESH_ACCOUNT_PROFILE", "work"),
        auth_token=os.environ.get("MESH_AUTH_TOKEN"),
    )

    worker = MeshWorker(config)

    def handle_shutdown(signum, frame):
        logger.info("Shutting down worker %s...", config.worker_id)
        worker.stop()
    signal.signal(signal.SIGTERM, handle_shutdown)
    signal.signal(signal.SIGINT, handle_shutdown)

    worker.start()


if __name__ == "__main__":
    run_worker()
```
